function calcDQI(acts){
  if(!acts.length) return 0;
  const total = acts.reduce((s,a)=>s+a.pts,0);
  return Math.min(100, Math.round(total / 3));
}

function getTier(score){
  return TIERS.find(t => score >= t.min && score <= t.max) || TIERS[0];
}

function fmtTime(ts){
  const d=new Date(ts);
  return d.getHours().toString().padStart(2,'0')+':'+d.getMinutes().toString().padStart(2,'0');
}

function renderStreakWeek(){
  const row = document.getElementById('streakRow');
  row.innerHTML = '';
  const today = new Date();
  for(let i=6;i>=0;i--){
    const d = new Date(today); d.setDate(today.getDate()-i);
    const ds = d.toDateString();
    const hasActs = state.activities.some(a=>new Date(a.ts).toDateString()===ds);
    const isToday = i===0;
    const div = document.createElement('div');
    div.className = 'streak-day ' + (isToday?'today':hasActs?'done':'miss');
    div.textContent = DAYS[d.getDay()];
    row.appendChild(div);
  }
}

function calcStreak(){
  if(!state.activities.length) return 0;
  let streak=0, d=new Date();
  d.setHours(0,0,0,0);
  while(true){
    const ds=d.toDateString();
    if(state.activities.some(a=>new Date(a.ts).toDateString()===ds)){ streak++; d.setDate(d.getDate()-1); }
    else break;
  }
  return streak;
}

    div.innerHTML=`<div class="wbar${isToday?' today':''}" style="height:4px" data-h="${h}px" title="${pts} pts"></div><div class="wday">${DAYS[d.getDay()]}</div>`;
    wrap.appendChild(div);
  }
  setTimeout(()=>{
    document.querySelectorAll('.wbar[data-h]').forEach(b=>{ b.style.height=b.dataset.h; });
  },200);
}

function renderCatBreakdown(){
  const el=document.getElementById('progCats');
  const totals={};
  state.activities.forEach(a=>{ totals[a.cat]=(totals[a.cat]||0)+a.pts; });
  const max=Math.max(...Object.values(totals),1);
  const sorted=CATS.map(c=>({...c,total:totals[c.id]||0})).filter(c=>c.total>0).sort((a,b)=>b.total-a.total);
  if(!sorted.length){ el.innerHTML='<div class="empty"><div class="empty-ic">📊</div><div class="empty-t">Нет данных</div><div class="empty-s">Начни логировать активности</div></div>'; return; }
  el.innerHTML=sorted.map(c=>`
    <div class="prog-cat-row">
      <div class="prog-cat-ic">${c.icon}</div>
      <div class="prog-cat-name">${c.name}</div>
      <div class="prog-bar-track"><div class="prog-bar-fill" data-w="${Math.round(c.total/max*100)}%"></div></div>
      <div class="prog-val">${c.total}б</div>
    </div>`).join('');
  setTimeout(()=>{
    document.querySelectorAll('.prog-bar-fill[data-w]').forEach(b=>{ b.style.width=b.dataset.w; });
  },200);
}

function showToast(msg,dur=2600){
  const t=document.getElementById('toast');
  t.textContent=msg; t.classList.add('show');
  setTimeout(()=>t.classList.remove('show'),dur);

function animNum(el,target,dec=0,dur=1000){
  const s=Date.now();
  const run=()=>{
    const p=Math.min((Date.now()-s)/dur,1);
    const e=1-Math.pow(1-p,3);
    el.textContent=(target*e).toFixed(dec);
    if(p<1) requestAnimationFrame(run);
  };
  run();
}
