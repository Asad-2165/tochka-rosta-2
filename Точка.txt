const canvas = document.getElementById('c');
const ctx = canvas.getContext('2d');
let W, H;
function resize() { W = canvas.width = window.innerWidth; H = canvas.height = window.innerHeight; }
resize();
window.addEventListener('resize', resize);

const N = 90;
const pts = Array.from({length: N}, () => ({
  x: Math.random() * W, y: Math.random() * H,
  vx: (Math.random() - .5) * .35, vy: (Math.random() - .5) * .35,
  r: Math.random() * 1.8 + .4,
  op: Math.random() * .45 + .1,
  col: Math.random() > .5 ? '0,102,255' : '0,229,255'
}));

let mouseX = W / 2, mouseY = H / 2;
document.addEventListener('mousemove', e => { mouseX = e.clientX; mouseY = e.clientY; });

function frame() {
  ctx.clearRect(0, 0, W, H);
  // radial glow
  const g = ctx.createRadialGradient(W/2, H/2, 0, W/2, H/2, W * .7);
  g.addColorStop(0, 'rgba(0,40,100,.22)');
  g.addColorStop(1, 'rgba(5,13,26,0)');
  ctx.fillStyle = g;
  ctx.fillRect(0, 0, W, H);
  // connections
  for (let i = 0; i < N; i++) {
    for (let j = i + 1; j < N; j++) {
      const dx = pts[i].x - pts[j].x, dy = pts[i].y - pts[j].y;
      const d = Math.sqrt(dx*dx + dy*dy);
      if (d < 130) {
        ctx.beginPath(); ctx.moveTo(pts[i].x, pts[i].y); ctx.lineTo(pts[j].x, pts[j].y);
        ctx.strokeStyle = `rgba(0,102,255,${(1 - d/130) * .13})`; ctx.lineWidth = .6; ctx.stroke();
      }
    }
  }
  // points + mouse interaction
  pts.forEach(p => {
    const mdx = mouseX - p.x, mdy = mouseY - p.y, md = Math.sqrt(mdx*mdx + mdy*mdy);
    if (md < 150) { p.vx -= mdx / md * .015; p.vy -= mdy / md * .015; }
    p.x += p.vx; p.y += p.vy;
    if (p.x < 0 || p.x > W) p.vx *= -1;
    if (p.y < 0 || p.y > H) p.vy *= -1;
    p.vx = Math.max(-1.2, Math.min(1.2, p.vx)); p.vy = Math.max(-1.2, Math.min(1.2, p.vy));
    ctx.beginPath(); ctx.arc(p.x, p.y, p.r, 0, Math.PI*2);
    ctx.fillStyle = `rgba(${p.col},${p.op})`; ctx.fill();
  });
  requestAnimationFrame(frame);
}
frame();

/* ===== COUNTERS ===== */
function animCount(el) {
  const t = parseFloat(el.dataset.cnt), dec = parseInt(el.dataset.dec);
  const dur = 2200, s = Date.now();
  const run = () => {
    const p = Math.min((Date.now()-s)/dur, 1);
    const e = 1 - Math.pow(1-p, 3);
    el.textContent = (t*e).toFixed(dec);
    if (p < 1) requestAnimationFrame(run);
  };
  run();
}
const cntObs = new IntersectionObserver(entries => {
  entries.forEach(en => { if (en.isIntersecting) { en.target.querySelectorAll('[data-cnt]').forEach(animCount); cntObs.unobserve(en.target); }});
}, {threshold: .5});
document.querySelectorAll('.hero-stats').forEach(el => cntObs.observe(el));
